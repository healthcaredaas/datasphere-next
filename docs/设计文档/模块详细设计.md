## 一、设计模式

### 1.1 使用的核心设计模式

| 模式 | 应用场景 | 实现位置 |
|------|----------|----------|
| **工厂模式** | 数据集成引擎创建 | DataEngineFactory |
| **策略模式** | 不同数据库类型的DDL生成 | DdlFactory |
| **模板方法模式** | 数据读取/写入流程 | AbstractReader/AbstractWriter |
| **责任链模式** | 数据转换处理链 | TransformChain |
| **观察者模式** | 任务执行状态监听 | JobEventListener |
| **单例模式** | 连接池管理 | ConnectionPoolManager |
| **建造者模式** | 复杂对象构建 | PipelineBuilder |

### 1.2 工厂模式 - 数据集成引擎

```java
/**
 * 数据集成引擎工厂
 */
@Component
public class DataEngineFactory {

    @Autowired
    private Map<String, DataEngine> engineMap;

    /**
     * 根据引擎类型获取引擎实例
     */
    public DataEngine getEngine(DiEngineType engineType) {
        String beanName = engineType.name().toLowerCase() + "Engine";
        DataEngine engine = engineMap.get(beanName);
        if (engine == null) {
            throw new UnsupportedOperationException("不支持的引擎类型: " + engineType);
        }
        return engine;
    }
}

/**
 * 引擎接口
 */
public interface DataEngine {
    /**
     * 引擎名称
     */
    String engineName();

    /**
     * 引擎配置转换
     */
    Config bindConfig(DataPipelineDto pipeline,
                      List<? extends Source> source,
                      List<? extends Sink> sink,
                      List<? extends Transform> transform);

    /**
     * 执行数据集成
     */
    DataResult execute(DataPipelineDto pipeline, Config config, JobInstance jobInstance);
}

/**
 * SeaTunnel引擎实现
 */
@Component("seatunnelEngine")
public class SeaTunnelEngine implements DataEngine {
    @Override
    public String engineName() {
        return "SeaTunnel";
    }

    @Override
    public Config bindConfig(DataPipelineDto pipeline, List<? extends Source> source,
                             List<? extends Sink> sink, List<? extends Transform> transform) {
        // 转换为SeaTunnel配置
        return SeaTunnelConfigBuilder.build(pipeline, source, sink, transform);
    }

    @Override
    public DataResult execute(DataPipelineDto pipeline, Config config, JobInstance jobInstance) {
        // 调用SeaTunnel执行
        return SeaTunnelExecutor.execute(config, jobInstance);
    }
}
```

### 1.3 策略模式 - DDL生成

```java
/**
 * DDL策略接口
 */
public interface DdlStrategy {
    String generateCreateTable(TableDefinition table);
    String generateAlterTable(AlterDefinition alter);
    String generateAddComment(CommentDefinition comment);
    String getDialect();
}

/**
 * MySQL DDL策略
 */
@Component
public class MysqlDdlStrategy implements DdlStrategy {
    @Override
    public String generateCreateTable(TableDefinition table) {
        StringBuilder sql = new StringBuilder("CREATE TABLE `");
        sql.append(table.getTableName()).append("` (\n");
        // 字段定义...
        sql.append("\n) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4");
        return sql.toString();
    }

    @Override
    public String getDialect() {
        return "mysql";
    }
}

/**
 * Oracle DDL策略
 */
@Component
public class OracleDdlStrategy implements DdlStrategy {
    @Override
    public String generateCreateTable(TableDefinition table) {
        // Oracle特定语法
        return "CREATE TABLE \"" + table.getTableName() + "\" (...)";
    }

    @Override
    public String getDialect() {
        return "oracle";
    }
}

/**
 * DDL策略工厂
 */
@Component
public class DdlStrategyFactory {
    @Autowired
    private List<DdlStrategy> strategies;

    public DdlStrategy getStrategy(String dialect) {
        return strategies.stream()
            .filter(s -> s.getDialect().equalsIgnoreCase(dialect))
            .findFirst()
            .orElseThrow(() -> new IllegalArgumentException("不支持的数据库类型: " + dialect));
    }
}
```

| FAILED | 执行失败 | 重试、编辑 |
| INACTIVE | 已下线 | 发布、删除 |

```java
    /**
     * 转换为SeaTunnel配置
     */
    public SeaTunnelConfig convertToSeaTunnel(DataPipelineDto pipeline) {
        SeaTunnelConfig config = new SeaTunnelConfig();

        // 1. 环境配置
        Env env = new Env();
        env.setJobMode(pipeline.getJobMode());
        env.setParallelism(pipeline.getParallelism());
        config.setEnv(env);

        // 2. Source配置转换

            .filter(c -> c.getConnectorType() == ConnectorType.SOURCE)
            .map(this::convertSource)
            .collect(Collectors.toList());
        config.setSource(sources);

        // 3. Transform配置转换
        List<SeaTunnelTransform> transforms = pipeline.getTransforms().stream()
            .map(this::convertTransform)
            .collect(Collectors.toList());
        config.setTransform(transforms);

        // 4. Sink配置转换
        List<SeaTunnelSink> sinks = pipeline.getConnectors().stream()
            .filter(c -> c.getConnectorType() == ConnectorType.SINK)
            .map(this::convertSink)
            .collect(Collectors.toList());
        config.setSink(sinks);

        return config;
    }

    private SeaTunnelSource convertSource(PipelineConnector connector) {
        SeaTunnelSource source = new SeaTunnelSource();
        source.setPluginName(connector.getPluginName());

        // 获取数据源配置
        DatasourceInfo dsInfo = datasourceService.getById(connector.getDatasourceId());
        DatasourceType dsType = datasourceTypeService.getById(dsInfo.getDsType());

        // 构建连接配置
        Map<String, Object> config = new HashMap<>();
        config.put("url", buildJdbcUrl(dsInfo));
        config.put("driver", dsType.getDriver());
        config.put("user", dsInfo.getUsername());
        config.put("password", decryptPassword(dsInfo.getPassword()));

        // 合并连接器特定配置

        source.setConfig(config);
```

```java
     * 执行质量检测

        if (checker == null) {
            throw new UnsupportedOperationException("不支持的规则类型: " + rule.getRuleType());
        }

        // 执行检测
        return checker.check(rule, context);
    }

    /**
     * 批量执行检测
     */
    public BatchCheckResult executeBatchCheck(List<QualityRule> rules, CheckContext context) {
        BatchCheckResult batchResult = new BatchCheckResult();

        for (QualityRule rule : rules) {
            try {
                CheckResult result = executeCheck(rule, context);
                batchResult.addResult(result);

                // 记录检测日志
                logCheckResult(rule, result);
            } catch (Exception e) {
                log.error("规则检测失败: ruleId={}, error={}", rule.getId(), e.getMessage());
                batchResult.addError(rule, e.getMessage());
            }
        }

        return batchResult;
    }
}

 * 完整性检查器

@Component
public class CompletenessChecker implements QualityChecker {

    @Override
    public RuleType getType() {
        return RuleType.COMPLETENESS;
    }

    @Override
    public CheckResult check(QualityRule rule, CheckContext context) {
        CompletenessConfig config = rule.getConfigAs(CompletenessConfig.class);
        String fieldName = config.getFieldName();

        // 构建检查SQL
        String sql = buildCheckSql(context, fieldName);

        // 执行统计
        CheckResult result = new CheckResult();
        result.setRuleId(rule.getId());
        result.setStartTime(LocalDateTime.now());

        try (Connection conn = context.getConnection();
             PreparedStatement ps = conn.prepareStatement(sql)) {

            ResultSet rs = ps.executeQuery();
            if (rs.next()) {
                long totalCount = rs.getLong("total_count");
                long nullCount = rs.getLong("null_count");
                long passCount = totalCount - nullCount;

                result.setTotalCount(totalCount);
                result.setPassCount(passCount);
                result.setFailCount(nullCount);
                result.setScore(calculateScore(passCount, totalCount));
                result.setStatus(nullCount == 0 ? CheckStatus.PASS : CheckStatus.FAIL);
            }
        } catch (SQLException e) {
            result.setStatus(CheckStatus.ERROR);
            result.setErrorMsg(e.getMessage());
        }

        result.setEndTime(LocalDateTime.now());
        return result;
    }

    private String buildCheckSql(CheckContext context, String fieldName) {
        return String.format(
            "SELECT COUNT(*) as total_count, " +
            "SUM(CASE WHEN %s IS NULL THEN 1 ELSE 0 END) as null_count " +
            "FROM %s",
            fieldName, context.getTableName()
        );
    }

    private double calculateScore(long passCount, long totalCount) {
        if (totalCount == 0) return 100.0;
        return Math.round((double) passCount / totalCount * 100 * 100) / 100.0;
    }
}
```

```java
        history.setVersion(current.getVersion());

        history.setCreateBy(current.getUpdateBy());
        history.setCreateTime(current.getUpdateTime());
        historyDao.insert(history);

        current.setVersion(current.getVersion() + 1);
        current.setData(newData);
        current.setStatus(MdmStatus.TO_AUDIT); // 新版本需要重新审批
        current.setUpdateBy(operator);
        current.setUpdateTime(LocalDateTime.now());
        current.setUpdateCount(current.getUpdateCount() + 1);

        return current;
    }

     * 回滚到指定版本
     */
    @Transactional
    public MasterDataInstance rollback(String instanceId, int targetVersion, String operator) {
        // 1. 获取目标版本历史
        MasterDataHistory history = historyDao.getByInstanceIdAndVersion(instanceId, targetVersion);
        if (history == null) {
            throw new DataNotFoundException("历史版本不存在: " + targetVersion);
        }

        // 2. 保存当前版本到历史
        MasterDataInstance current = instanceDao.getById(instanceId);
        MasterDataHistory currentHistory = new MasterDataHistory();
        BeanUtils.copyProperties(current, currentHistory);
        currentHistory.setId(null);
        historyDao.insert(currentHistory);

        // 3. 恢复到目标版本
        current.setData(history.getData());
        current.setVersion(current.getVersion() + 1); // 版本号+1
        current.setStatus(MdmStatus.TO_AUDIT);
        current.setUpdateBy(operator);
        current.setUpdateTime(LocalDateTime.now());
        instanceDao.updateById(current);

        return current;
    }
```

### 6.4 HIE消息路由逻辑

```java
/**
 * HIE消息路由器
 */
@Component
public class HieMessageRouter {

    @Autowired
    private InteractionServiceDao serviceDao;

    @Autowired
    private SubscriptionDao subscriptionDao;

    @Autowired
    private RocketMQTemplate rocketMQTemplate;

    /**
     * 路由消息
     */
    public RouteResult route(Message message) {
        RouteResult result = new RouteResult();

        // 1. 解析消息头
        MshHeader header = parseMshHeader(message);

        // 2. 查找匹配的服务
        List<InteractionService> services = serviceDao.findByMshConfig(header);
        if (services.isEmpty()) {
            result.setSuccess(false);
            result.setErrorMsg("未找到匹配的服务");
            return result;
        }

        // 3. 查找订阅关系
        for (InteractionService service : services) {
            List<Subscription> subscriptions = subscriptionDao.findByServiceId(service.getId());

            for (Subscription subscription : subscriptions) {
                // 4. 验证消息过滤条件
                if (!matchFilter(message, subscription.getFilterConfig())) {
                    continue;
                }

                // 5. 路由到订阅者
                RouteTarget target = new RouteTarget();
                target.setSubscriptionId(subscription.getId());
                target.setSubscriberCode(subscription.getSubscriberCode());
                target.setEndpoint(subscription.getEndpoint());
                target.setServiceType(subscription.getServiceType());

                // 6. 发送消息
                boolean sent = sendMessage(message, target);
                target.setSuccess(sent);

                result.addTarget(target);
            }
        }

        result.setSuccess(!result.getTargets().isEmpty());
        return result;
    }

    /**
     * 解析MSH消息头
     */
    private MshHeader parseMshHeader(Message message) {
        MshHeader header = new MshHeader();

        // 从消息中提取9个标准字段
        header.setMsh1(message.getHeader("MSH1")); // 消息类型
        header.setMsh2(message.getHeader("MSH2")); // 触发事件
        header.setMsh3(message.getHeader("MSH3")); // 发送系统
        header.setMsh4(message.getHeader("MSH4")); // 发送机构
        header.setMsh5(message.getHeader("MSH5")); // 接收系统
        header.setMsh6(message.getHeader("MSH6")); // 接收机构
        header.setMsh7(message.getHeader("MSH7")); // 消息时间
        header.setMsh8(message.getHeader("MSH8")); // 消息ID
        header.setMsh9(message.getHeader("MSH9")); // 处理ID

        return header;
    }

    /**
     * 匹配过滤条件
     */
    private boolean matchFilter(Message message, FilterConfig filterConfig) {
        if (filterConfig == null) {
            return true;
        }

        // XPath过滤
        if (StringUtils.isNotBlank(filterConfig.getXpath())) {
            String value = XmlUtils.extractByXpath(message.getBody(), filterConfig.getXpath());
            if (!filterConfig.getExpectedValue().equals(value)) {
                return false;
            }
        }

        // JSONPath过滤
        if (StringUtils.isNotBlank(filterConfig.getJsonPath())) {
            String value = JsonUtils.extractByJsonPath(message.getBody(), filterConfig.getJsonPath());
            if (!filterConfig.getExpectedValue().equals(value)) {
                return false;
            }
        }

        return true;
    }

    /**
     * 发送消息
     */
    private boolean sendMessage(Message message, RouteTarget target) {
        try {
            switch (target.getServiceType()) {
                case MQ:
                    rocketMQTemplate.asyncSend(target.getEndpoint(), message, new SendCallback() {
                        @Override
                        public void onSuccess(SendResult sendResult) {
                            log.info("消息发送成功: target={}", target.getSubscriberCode());
                        }
                        @Override
                        public void onException(Throwable e) {
                            log.error("消息发送失败: target={}, error={}", target.getSubscriberCode(), e.getMessage());
                        }
                    });
                    return true;
                case HTTP:
                    // HTTP调用
                    return httpSend(message, target.getEndpoint());
                case SOAP:
                    // SOAP调用
                    return soapSend(message, target.getEndpoint());
                default:
                    return false;
            }
        } catch (Exception e) {
            log.error("消息路由失败: target={}, error={}", target.getSubscriberCode(), e.getMessage());
            return false;
        }
    }
}
```
